140  void compute_2D(const T* fd, const T* fr, const RGrid &rgrid){
141 
142  if(rgrid.dim != 2){
143  printf(" HHD_potentials::compute_2D requires a 2D regular grid!\n");
144  exit(1);
145  }
146 
147  printf(" Solving 2D natural HHD...");
148  fflush(stdout);
149 
150  clock_t tic = clock();
151 
152  // variables for showing progress
153  float percent_factor = 100.0f / (float)rgrid.sz;
154  float percent_unit = 0.1; // show progress at this percent work
155  uint progress_unit = std::max((uint)1, (uint) (percent_unit / percent_factor));
156 
157  //printf(" %f %f %d\n", percent_factor, percent_unit, progress_unit);
158  // -------------------------------------------------------------------
159  // the 2D Green's function is ---- 1/(2pi) log( dist(x,x_0) )
160  // instead, I use ------------ 1/(4pi) log( dist^2(x,x_0) )
161  static T Greens_factor = 0.25/M_PI;
162 
163  // output and temporary memory
164 
165  D = new T[rgrid.sz];
166  Ru = new T[rgrid.sz];
167 
168  T* fdt = new T[rgrid.sz];
169  T* frt = new T[rgrid.sz];
170 
171  // -------------------------------------------------------------------
172  // compute at v0
173  for(size_t v0 = 0; v0 < rgrid.sz; v0++){
174 
175  // populate the temporary field for the entire grid!
176  #pragma omp parallel for
177  for(size_t v = 0; v < rgrid.sz; v++){
178 
179  // don't want to compute log (0)
180  T dist = (v == v0) ? (0.01 * (rgrid.dx+rgrid.dy)) :
181  rgrid.get_dist_sq(v, v0);
182 
183  fdt[v] = (log(dist) * fd[v]);
184  frt[v] = (log(dist) * fr[v]);
185  }
186 
187  // integrate
188  D[v0] = Greens_factor * integ::trapezoidal_2D(fdt, rgrid.X, rgrid.Y, rgrid.dx, rgrid.dy);
189  Ru[v0] = Greens_factor * integ::trapezoidal_2D(frt, rgrid.X, rgrid.Y, rgrid.dx, rgrid.dy);
190 
191  if(v0 % progress_unit == 0){
192  printf("\r Solving 2D natural HHD... %.1f %%", (float)v0*percent_factor);
193  fflush(stdout);
194  }
195  }
196 
197  delete []fdt;
198  delete []frt;
199 
200  double sec = (double)(clock() - tic)/CLOCKS_PER_SEC;
201  if(sec < 1) printf("\r Solving 2D natural HHD... Done! in %.3f msec.\n", 1000.0*sec);
202  else printf("\r Solving 2D natural HHD... Done! in %.3f sec.\n", sec);
203  }
