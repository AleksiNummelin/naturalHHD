233  float get_dist_sq(size_t v1, size_t v2) const{
234 
235  if(dim == 1){
236  long int dist = (v1-v2);
237  return dist*dist*sq_dx;
238  }
239  if(dim == 2){
240 
241  size_t x1, y1, x2, y2;
242  get_coords(v1, x1, y1);
243  get_coords(v2, x2, y2);
244 
245  long int distx = (x1-x2);
246  long int disty = (y1-y2);
247 
248  return (distx*distx*sq_dx) + (disty*disty*sq_dy);
249  }
250  if(dim == 3){
251 
252  size_t x1, y1, z1, x2, y2, z2;
253  get_coords(v1, x1, y1, z1);
254  get_coords(v2, x2, y2, z2);
255 
256  long int distx = (x1-x2);
257  long int disty = (y1-y2);
258  long int distz = (z1-z2);
259 
260  return (distx*distx*sq_dx) + (disty*disty*sq_dy) + (distz*distz*sq_dz);
261  }
262 
263  return 0.0;
264  }
